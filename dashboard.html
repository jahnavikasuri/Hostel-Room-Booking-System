window.onload = async () => { const res = await fetch('http://localhost:5000/api/bookings/all'); const bookings = await res.json(); const table = document.getElementById('bookingTable'); bookings.forEach(b => { const row = table.insertRow(); row.innerHTML = `${b.name}${b.roomType}${new Date(b.date).toLocaleDateString()}`; }); };
